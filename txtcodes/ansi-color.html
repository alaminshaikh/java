            public class color {
                // ANSI escape codes for text colors
                public static final String ANSI_RESET = "\u001B[0m";
                public static final String ANSI_RED = "\u001B[31m";
                public static final String ANSI_GREEN = "\u001B[32m";
                public static final String ANSI_YELLOW = "\u001B[33m";
                public static final String ANSI_BLUE = "\u001B[34m";
                public static final String ANSI_PURPLE = "\u001B[41m"; //red background
                public static final String ANSI_CYAN = "\u001B[44m";    //blue background
                public static final String ANSI_WHITE = "\u001B[37m";
            
                public static void main(String[] args) {
                    System.out.println(ANSI_RED + "Hello, this text is red!" + ANSI_RESET);
                    System.out.println(ANSI_GREEN + "Hello, this text is green!" + ANSI_RESET);
                    System.out.println(ANSI_YELLOW + "Hello, this text is yellow!" + ANSI_RESET);
                    System.out.println(ANSI_BLUE + "Hello, this text is blue!" + ANSI_RESET);
                    System.out.println(ANSI_PURPLE + "Hello, this text is RED HIGHLIGHTED!" + ANSI_RESET);
                    System.out.println(ANSI_CYAN + "Hello, this text is BLUE HIGHLIGHTED!" + ANSI_RESET);
                    System.out.println(ANSI_WHITE + "Hello, this text is white!" + ANSI_RESET);
                    System.out.println("\u001B[4m\u001B[44mHey Underline\u001B[0m");
                    System.out.println("\u001B[1m\u001B[44mHey Bold\u001B[0m");
                    System.out.println("\u001B[30m\u001B[107mHey Black text White Background\u001B[0m");
                }
